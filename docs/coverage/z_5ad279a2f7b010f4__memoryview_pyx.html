24cdef class StridedMemoryView: 

25 """A class holding metadata of a strided dense array/tensor. 

26  

27 A :obj:`StridedMemoryView` instance can be created in three ways: 

28  

29 1. Using the :obj:`args_viewable_as_strided_memory` decorator (recommended) 

30 2. Explicit construction relying on DLPack or CUDA Array Interface, see below. 

31 3. From :obj:`~_memory.Buffer` and a :obj:`_StridedLayout` (see :meth:`from_buffer` classmethod) 

32  

33 ``StridedMemoryView(obj, stream_ptr)`` can be used to create a view from 

34 objects supporting either DLPack (up to v1.0) or CUDA Array Interface 

35 (CAI) v3. When wrapping an arbitrary object it will try the DLPack protocol 

36 first, then the CAI protocol. A :obj:`BufferError` is raised if neither is 

37 supported. 

38  

39 Since either way would take a consumer stream, for DLPack it is passed to 

40 ``obj.__dlpack__()`` as-is (except for :obj:`None`, see below); for CAI, a 

41 stream order will be established between the consumer stream and the 

42 producer stream (from ``obj.__cuda_array_interface__()["stream"]``), as if 

43 ``cudaStreamWaitEvent`` is called by this method. 

44  

45 To opt-out of the stream ordering operation in either DLPack or CAI, 

46 please pass ``stream_ptr=-1``. Note that this deviates (on purpose) 

47 from the semantics of ``obj.__dlpack__(stream=None, ...)`` since ``cuda.core`` 

48 does not encourage using the (legacy) default/null stream, but is 

49 consistent with the CAI's semantics. For DLPack, ``stream=-1`` will be 

50 internally passed to ``obj.__dlpack__()`` instead. 

51  

52 Parameters 

53 ---------- 

54 obj : Any 

55 Any objects that supports either DLPack (up to v1.0) or CUDA Array 

56 Interface (v3). 

57 stream_ptr: int 

58 The pointer address (as Python `int`) to the **consumer** stream. 

59 Stream ordering will be properly established unless ``-1`` is passed. 

60  

61  

62 Attributes 

63 ----------- 

64 ptr : int 

65 Pointer to the tensor buffer (as a Python `int`). 

66 device_id : int 

67 The device ID for where the tensor is located. It is -1 for CPU tensors 

68 (meaning those only accessible from the host). 

69 is_device_accessible : bool 

70 Whether the tensor data can be accessed on the GPU. 

71 readonly: bool 

72 Whether the tensor data can be modified in place. 

73 exporting_obj : Any 

74 A reference to the original tensor object that is being viewed. 

75 If the view is created with :meth:`from_buffer`, 

76 it will be the Buffer instance passed to the method. 

77

412cdef StridedMemoryView view_as_dlpack(obj, stream_ptr, view=None): 

413 cdef int dldevice, device_id 

414 cdef bint is_device_accessible, is_readonly 

415 is_device_accessible = False 

416 dldevice, device_id = obj.__dlpack_device__() 

417 if dldevice == _kDLCPU: 

418 assert device_id == 0 

419 device_id = -1 

420 if stream_ptr is None: 

421 raise BufferError("stream=None is ambiguous with view()") 

422 elif stream_ptr == -1: 

423 stream_ptr = None 

424 elif dldevice == _kDLCUDA: 

425 assert device_id >= 0

427 # no need to check other stream values, it's a pass-through 

428 if stream_ptr is None:

431 is_device_accessible = True

433 # accessed on both host and device

565cpdef StridedMemoryView view_as_cai(obj, stream_ptr, view=None): 

566 cdef dict cai_data = obj.__cuda_array_interface__ 

567 if cai_data["version"] < 3: 

568 raise BufferError("only CUDA Array Interface v3 or above is supported") 

569 if cai_data.get("mask") is not None: 

570 raise BufferError("mask is not supported") 

571 if stream_ptr is None: 

572 raise BufferError("stream=None is ambiguous with view()") 

573  

574 cdef StridedMemoryView buf = StridedMemoryView() if view is None else view 

575 buf.exporting_obj = obj 

576 buf.metadata = cai_data 

577 buf.dl_tensor = NULL 

578 buf.ptr, buf.readonly = cai_data["data"] 

579 buf.is_device_accessible = True 

580 buf.device_id = handle_return( 

581 driver.cuPointerGetAttribute( 

582 driver.CUpointer_attribute.CU_POINTER_ATTRIBUTE_DEVICE_ORDINAL, 

583 buf.ptr)) 

584  

585 cdef intptr_t producer_s, consumer_s 

586 stream_ptr = int(stream_ptr) 

587 if stream_ptr != -1: 

588 stream = cai_data.get("stream") 

589 if stream is not None: 

590 producer_s = <intptr_t>(stream) 

591 consumer_s = <intptr_t>(stream_ptr) 

592 assert producer_s > 0 

593 # establish stream order 

594 if producer_s != consumer_s: 

595 e = handle_return(driver.cuEventCreate( 

596 driver.CUevent_flags.CU_EVENT_DISABLE_TIMING)) 

597 handle_return(driver.cuEventRecord(e, producer_s)) 

598 handle_return(driver.cuStreamWaitEvent(consumer_s, e, 0)) 

599 handle_return(driver.cuEventDestroy(e)) 

600  

601 return buf

610 implementation, while keeping the function body array-library-agnostic (if 

611 desired). 

612  

613 Inside the decorated function, the specified arguments become instances 

614 of an (undocumented) proxy type, regardless of its original source. A 

615 :obj:`StridedMemoryView` instance can be obtained by passing the (consumer) 

616 stream pointer (as a Python `int`) to the proxies's ``view()`` method. For 

617 example: 

618  

619 .. code-block:: python 

620  

621 @args_viewable_as_strided_memory((1,)) 

622 def my_func(arg0, arg1, arg2, stream: Stream): 

623 # arg1 can be any object supporting DLPack or CUDA Array Interface 

624 view = arg1.view(stream.handle) 

625 assert isinstance(view, StridedMemoryView) 

626 ... 

627  

628 Parameters 

629 ---------- 

630 arg_indices : tuple 

631 The indices of the target positional arguments. 

632 """ 

633 def wrapped_func_with_indices(func): 

634 @functools.wraps(func) 

635 def wrapped_func(*args, **kwargs): 

636 args = list(args) 

637 cdef int idx 

638 for idx in arg_indices: 

639 args[idx] = _StridedMemoryViewProxy(args[idx]) 

640 return func(*args, **kwargs) 

641 return wrapped_func 

642 return wrapped_func_with_indices

645cdef inline _StridedLayout layout_from_dlpack(DLTensor* dl_tensor): 

646 cdef _StridedLayout layout = _StridedLayout.__new__(_StridedLayout) 

647 cdef int nbits = dl_tensor.dtype.bits * dl_tensor.dtype.lanes 

648 cdef int itemsize = nbits >> 3 

649 if (itemsize << 3) != nbits: 

650 raise ValueError("dl_tensor.dtype.bits must be a multiple of 8") 

651 layout.init_from_ptr(dl_tensor.ndim, dl_tensor.shape, dl_tensor.strides, itemsize) 

652 return layout

655cdef _StridedLayout layout_from_cai(object metadata): 

656 cdef _StridedLayout layout = _StridedLayout.__new__(_StridedLayout) 

657 cdef object shape = metadata["shape"] 

658 cdef object strides = metadata.get("strides") 

659 cdef int itemsize = _typestr2itemsize[metadata["typestr"]] 

660 layout.init_from_tuple(shape, strides, itemsize, True) 

661 return layout

664cdef inline intptr_t get_data_ptr(object buffer, _StridedLayout layout) except? 0: 

665 return <intptr_t>(int(buffer.handle)) + layout.get_slice_offset_in_bytes()

668cdef inline int view_buffer_strided( 

669 StridedMemoryView view, 

670 object buffer, 

671 _StridedLayout layout, 

672 object dtype, 

673 bint is_readonly, 

674) except -1: 

675 if dtype is not None: 

676 dtype = numpy.dtype(dtype) 

677 if dtype.itemsize != layout.itemsize: 

678 raise ValueError( 

679 f"The dtype's itemsize ({dtype.itemsize}) does not match the layout's " 

680 f"itemsize ({layout.itemsize}). Please use :meth:`_StridedLayout.repacked` " 

681 f"to transform the layout to the desired itemsize." 

682 ) 

683 # Check the layout's offset range [min_offset, max_offset] fits 

684 # within the [0, buffer.size - 1] range. 

685 # The required_size_in_bytes fails if min_offset < 0. 

686 # NB. For external memory, both positive and negative offsets can be valid, 

687 # but for a proper check we'd need to know both size and data offset, 

688 # while neither is reported by the packages. 

689 cdef bint is_allocated = buffer.memory_resource is not None 

690 if is_allocated and buffer.size < layout.get_required_size_in_bytes(): 

691 raise ValueError( 

692 f"Buffer size is too small for the layout. " 

693 f"Expected at least {layout.get_required_size_in_bytes()} bytes, " 

694 f"got {buffer.size} bytes." 

695 ) 

696 # set the public attributes 

697 view.ptr = get_data_ptr(buffer, layout) 

698 view.device_id = buffer.device_id 

699 view.is_device_accessible = buffer.is_device_accessible 

700 view.readonly = is_readonly 

701 view.exporting_obj = view._buffer = buffer 

702 # no dlpack/cai metadata 

703 view.dl_tensor = NULL 

704 view.metadata = None 

705 # we get the layout from the caller 

706 view._layout = layout 

707 view._dtype = dtype 

708 return 0
