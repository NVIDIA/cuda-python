58def add_dll_directory(dll_abs_path: str) -> None: 

59 """Add a DLL directory to the search path and update PATH environment variable. 

60 

61 Args: 

62 dll_abs_path: Absolute path to the DLL file 

63 

64 Raises: 

65 AssertionError: If the directory containing the DLL does not exist 

66 """ 

67 dirpath = os.path.dirname(dll_abs_path) 

68 assert os.path.isdir(dirpath), dll_abs_path 

69 

70 # Add the DLL directory to the search path 

71 result = kernel32.AddDllDirectory(dirpath) 

72 if not result: 

73 # Fallback: just update PATH if AddDllDirectory fails 

74 pass 

75 

76 # Update PATH as a fallback for dependent DLL resolution 

77 curr_path = os.environ.get("PATH") 

78 os.environ["PATH"] = dirpath if curr_path is None else os.pathsep.join((curr_path, dirpath))

81def abs_path_for_dynamic_library(libname: str, handle: ctypes.wintypes.HMODULE) -> str: 

82 """Get the absolute path of a loaded dynamic library on Windows.""" 

83 # Create buffer for the path 

84 buffer = ctypes.create_unicode_buffer(260) # MAX_PATH 

85 length = kernel32.GetModuleFileNameW(handle, buffer, len(buffer)) 

86 

87 if length == 0: 

88 error_code = ctypes.GetLastError() # type: ignore[attr-defined] 

89 raise RuntimeError(f"GetModuleFileNameW failed for {libname!r} (error code: {error_code})") 

90 

91 # If buffer was too small, try with larger buffer 

92 if length == len(buffer): 

93 buffer = ctypes.create_unicode_buffer(32768) # Extended path length 

94 length = kernel32.GetModuleFileNameW(handle, buffer, len(buffer)) 

95 if length == 0: 

96 error_code = ctypes.GetLastError() # type: ignore[attr-defined] 

97 raise RuntimeError(f"GetModuleFileNameW failed for {libname!r} (error code: {error_code})") 

98 

99 return buffer.value

136 """Load a dynamic library from the given path. 

137

141

144

158 return LoadedDL(found_path, False, ctypes_handle_to_unsigned_int(handle), found_via)
