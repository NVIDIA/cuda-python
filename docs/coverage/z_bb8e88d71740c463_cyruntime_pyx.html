1014cdef cudaError_t getLocalRuntimeVersion(int* runtimeVersion) except ?cudaErrorCallRequiresNewerDriver nogil: 

1015 # Load 

1016 with gil: 

1017 loaded_dl = load_nvidia_dynamic_lib("cudart") 

1018 handle = <void *><uintptr_t>loaded_dl._handle_uint 

1019 __cudaRuntimeGetVersion = dlfcn.dlsym(handle, 'cudaRuntimeGetVersion') 

1020  

1021 if __cudaRuntimeGetVersion == NULL: 

1022 with gil: 

1023 raise RuntimeError(f'Function "cudaRuntimeGetVersion" not found in {loaded_dl.abs_path}') 

1024  

1025 # Call 

1026 cdef cudaError_t err = cudaSuccess 

1027 err = (<cudaError_t (*)(int*) except ?cudaErrorCallRequiresNewerDriver nogil> __cudaRuntimeGetVersion)(runtimeVersion) 

1028  

1029 # We explicitly do *NOT* cleanup the library handle here, acknowledging 

1030 # that, yes, the handle leaks. The reason is that there's a 

1031 # `functools.cache` on the top-level caller of this function. 

1032 # 

1033 # This means this library would be opened once and then immediately closed, 

1034 # all the while remaining in the cache lurking there for people to call. 

1035 # 

1036 # Since we open the library one time (technically once per unique library name), 

1037 # there's not a ton of leakage, which we deem acceptable for the 1000x speedup 

1038 # achieved by caching (ultimately) `ctypes.CDLL` calls. 

1039 # 

1040 # Long(er)-term we can explore cleaning up the library using higher-level 

1041 # Python mechanisms, like `__del__` or `weakref.finalizer`s. 

1042  

1043 return err
